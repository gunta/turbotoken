# turbotoken -- Upstream Sync Strategy

> How we track, clone, and adapt tests from upstream tokenizer projects.
> Ensures turbotoken stays byte-perfect compatible as upstreams evolve.

---

## Overview

We maintain synced copies of upstream tokenizer repos for:
1. **Correctness oracle** -- tiktoken IS the definition of correct
2. **Test suite reuse** -- adapt their tests to verify our output
3. **Algorithm reference** -- study their implementations
4. **Version tracking** -- detect API changes early

---

## Upstream Repos

### 1. tiktoken (OpenAI) -- THE ORACLE

| Property | Value |
|----------|-------|
| **Repo** | https://github.com/openai/tiktoken |
| **Local path** | `upstream/tiktoken/` |
| **Sync method** | git submodule (shallow clone) |
| **What we use** | Test suite, encoding definitions, API surface |
| **Sync frequency** | Weekly + on every release |

**Files we care about:**
```
upstream/tiktoken/
+-- tests/
|   +-- test_encoding.py          # MAIN: byte-perfect correctness tests
|   +-- test_misc.py              # Edge cases
+-- tiktoken/
|   +-- core.py                   # API surface to replicate
|   +-- model.py                  # Model -> encoding mapping
+-- tiktoken_ext/
    +-- openai_public.py          # Encoding definitions (merge table URLs)
```

**How we adapt their tests:**
```python
# tests/upstream/test_tiktoken_compat.py
# Auto-generated by scripts/sync-upstream.ts

import turbotoken
import tiktoken

# For every test in tiktoken's test suite, run it with both
# and assert identical output

def test_simple_encoding():
    """Adapted from tiktoken/tests/test_encoding.py"""
    tt = turbotoken.get_encoding("cl100k_base")
    tk = tiktoken.get_encoding("cl100k_base")
    assert tt.encode("hello world") == tk.encode("hello world")
```

## Sync Script

```typescript
// scripts/sync-upstream.ts
// Run: bun run scripts/sync-upstream.ts

import { $ } from "bun";

interface Upstream {
  name: string;
  repo: string;
  branch: string;
  sparse?: string[];  // sparse checkout paths (if we don't need everything)
}

const upstreams: Upstream[] = [
  {
    name: "tiktoken",
    repo: "openai/tiktoken",
    branch: "main",
    sparse: ["tests/", "tiktoken/core.py", "tiktoken/model.py", "tiktoken_ext/"],
  },
  {
    name: "rs-bpe",
    repo: "gweidart/rs-bpe",
    branch: "main",
    sparse: ["tests/", "src/"],
  },
  {
    name: "github-bpe",
    repo: "github/bpe",
    branch: "main",
    sparse: ["tests/", "src/"],
  },
  {
    name: "compare-tokenizers",
    repo: "transitive-bullshit/compare-tokenizers",
    branch: "main",
  },
  {
    name: "gpt-tokenizer",
    repo: "niieani/gpt-tokenizer",
    branch: "main",
  },
];

async function syncUpstream(upstream: Upstream) {
  const dir = `upstream/${upstream.name}`;

  try {
    // Check if already cloned
    const gitDir = Bun.file(`${dir}/.git`);
    if (await gitDir.exists()) {
      console.log(`Updating ${upstream.name}...`);
      await $`cd ${dir} && git fetch origin ${upstream.branch} && git reset --hard origin/${upstream.branch}`;
    } else {
      console.log(`Cloning ${upstream.name}...`);
      await $`git clone --depth 1 --branch ${upstream.branch} https://github.com/${upstream.repo}.git ${dir}`;
    }
    console.log(`  Synced ${upstream.name}`);
  } catch (e) {
    console.error(`  FAILED to sync ${upstream.name}: ${e}`);
  }
}

// Sync all upstreams
console.log("=== Syncing upstream repos ===\n");
for (const upstream of upstreams) {
  await syncUpstream(upstream);
}

// Adapt tiktoken tests
console.log("\n=== Adapting tiktoken tests ===\n");
await $`mkdir -p tests/upstream`;

// Copy tiktoken tests with a header noting they're auto-synced
const header = `# AUTO-GENERATED by scripts/sync-upstream.ts
# Do not edit directly. Changes will be overwritten on next sync.
# Source: upstream/tiktoken/tests/
# Run: bun run scripts/sync-upstream.ts

`;

try {
  const testFile = await Bun.file("upstream/tiktoken/tests/test_encoding.py").text();
  await Bun.write("tests/upstream/test_tiktoken_compat.py", header + testFile);
  console.log("  Adapted tiktoken test_encoding.py");
} catch (e) {
  console.log("  Skipping tiktoken test adaptation (file not found)");
}

console.log("\n=== Sync complete ===");
```

## Version Tracking

### How We Detect Upstream Changes

```typescript
// scripts/check-upstream-versions.ts
// Run: bun run scripts/check-upstream-versions.ts
// CI: runs weekly, opens issue if major changes detected

import { $ } from "bun";

// Check tiktoken PyPI version
const tiktokenVersion = await $`pip index versions tiktoken 2>/dev/null | head -1`.text();
console.log(`tiktoken latest: ${tiktokenVersion.trim()}`);

// Check if tiktoken API surface changed
const currentCore = await Bun.file("upstream/tiktoken/tiktoken/core.py").text();
const savedHash = await Bun.file(".upstream-hashes/tiktoken-core.txt").text().catch(() => "");
const newHash = Bun.hash(currentCore).toString();

if (savedHash.trim() !== newHash) {
  console.log("WARNING: tiktoken core.py has changed since last check!");
  console.log("Run: bun run scripts/sync-upstream.ts && bun run scripts/diff-api.ts");
}

await Bun.write(".upstream-hashes/tiktoken-core.txt", newHash);
```

## Test Adaptation Rules

### From tiktoken Python tests -> turbotoken Python tests

| tiktoken | turbotoken | Notes |
|----------|-----------|-------|
| `import tiktoken` | `import turbotoken` | Direct swap |
| `tiktoken.get_encoding(x)` | `turbotoken.get_encoding(x)` | Same API |
| `tiktoken.encoding_for_model(x)` | `turbotoken.encoding_for_model(x)` | Same API |
| Internal implementation tests | SKIP | We don't replicate internals |
| Performance assertions | ADAPT | Our targets are different |

### From GitHub bpe Rust tests -> turbotoken Python tests

| Rust test | Python adaptation | Notes |
|-----------|------------------|-------|
| `#[test] fn test_encode()` | `def test_encode():` | Translate assertions |
| `assert_eq!(result, expected)` | `assert result == expected` | Direct |
| `b"hello"` | `b"hello"` | Same in Python |
| Rust-specific tests | SKIP | Not relevant |

### From compare-tokenizers JS tests -> turbotoken JS tests

| JS test | turbotoken JS test | Notes |
|---------|-------------------|-------|
| `import { encoding } from "tiktoken"` | `import { getEncoding } from "turbotoken"` | API mapping |
| `expect(result).toEqual(expected)` | Same (Jest/Vitest) | Direct |
| Browser-specific tests | ADAPT | For our WASM build |

## CI Integration

```yaml
# .github/workflows/upstream-check.yml
name: Upstream Version Check
on:
  schedule:
    - cron: "0 9 * * 1"  # Every Monday at 9am UTC
  workflow_dispatch:

jobs:
  check:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          submodules: true
      - uses: oven-sh/setup-bun@v1
      - run: bun run scripts/sync-upstream.ts
      - run: bun run scripts/check-upstream-versions.ts
      - name: Check for changes
        run: |
          if git diff --quiet upstream/; then
            echo "No upstream changes"
          else
            echo "::warning::Upstream repos have changes. Review needed."
            git diff --stat upstream/
          fi
```
